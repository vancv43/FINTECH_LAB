Slide 5 · Lý thuyết nền tảng trước khi code
🧠
Nguyên lý cốt lõi (Slide 5): Chatbot chỉ là User → LLM → Answer. AI Agent thật sự là một pipeline đa lớp. Mọi code trong lab này đều hiện thực hóa pipeline đó.
❌ Chatbot (đơn giản)
  • User Input → LLM → Answer
  • Không có memory giữa các lần hỏi
  • Không dùng được công cụ bên ngoài
  • Không có reasoning chain
  • Ví dụ: ChatGPT không có plugin
✅ AI Agent (đa lớp)
  • Retrieval (FAISS/TF-IDF) → Knowledge
  • Reasoning (LLM Chain) → Logic
  • Prediction (RandomForest) → Numbers
  • Memory (JSON/DB) → Context
  • LLM Explainer → Human output

"""
Semantic Knowledge Base
Layer 1 của AI Agent Pipeline (Slide 5, 12, 13)
Dùng TF-IDF + cosine similarity để tìm pattern gần nhất.
Tương đương FAISS vector search nhưng không cần GPU.
"""

import os
import pandas as pd
import numpy as np
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.metrics.pairwise import cosine_similarity

# ── Config ──────────────────────────────────────────────
DATA_PATH = os.path.join(os.path.dirname(__file__), "..", "data", "enterprise_knowledge.csv")


class EnterpriseKnowledgeBase:
    """
    TF-IDF vector store — đại diện cho FAISS Knowledge Base (Slide 12).
    Có thể nâng cấp sang FAISS hoặc sentence-transformers khi có GPU.
    """

    def __init__(self):
        self.df = pd.read_csv(DATA_PATH)
        self.vectorizer = TfidfVectorizer(
            ngram_range=(1, 2),   # unigram + bigram
            stop_words="english",
            max_features=5000,
        )
        corpus = self.df["issue_pattern"].tolist()
        self.index_matrix = self.vectorizer.fit_transform(corpus)
        print(f"[KB] Loaded {len(corpus)} patterns from knowledge base")

    def search(self, query: str, top_k: int = 1) -> list[dict]:
        """
        Tìm top_k pattern gần nhất với query.
        Returns: list of dicts với confidence_score.
        """
        q_vec = self.vectorizer.transform([query])
        scores = cosine_similarity(q_vec, self.index_matrix)[0]
        top_indices = np.argsort(scores)[::-1][:top_k]
        results = []
        for idx in top_indices:
            row = self.df.iloc[idx]
            results.append({
                "issue_pattern":     row["issue_pattern"],
                "solution":         row["solution"],
                "department":       row["department"],
                "urgency_level":    row["urgency_level"],
                "recommended_action": row["recommended_action"],
                "confidence_score":  float(scores[idx]),
            })
        return results

    def get_all_issues(self) -> list[str]:
        return self.df["issue_pattern"].tolist()

    def get_stats(self) -> dict:
        return {
            "total_patterns": len(self.df),
            "departments":    self.df["department"].nunique(),
            "urgency_distribution": self.df["urgency_level"].value_counts().to_dict(),
        }


# ── Quick test ──────────────────────────────────────────

    results = kb.search("KPI performance delays for employees")
    for r in results:
        print(f"Dept: {r['department']}")
        print(f"Confidence: {r['confidence_score']:.2%}")
        print(f"Action: {r['recommended_action']}")

"""
Predictive Budget Engine — Layer 3 của AI Agent Pipeline
RandomForestRegressor: predict deployment cost từ project parameters.
Tương đương DBS ALAN credit scoring engine (Slide 8, 11, 13).
"""

import os
import numpy as np
import pandas as pd
from sklearn.ensemble import RandomForestRegressor
from sklearn.preprocessing import LabelEncoder
from sklearn.metrics import r2_score

DATA_PATH = os.path.join(os.path.dirname(__file__), "..", "data", "project_cost_data.csv")
RISK_MAP  = {"low": 1, "medium": 2, "high": 3}
REGIONS   = ["northeast", "northwest", "southeast", "southwest"]


class BudgetPredictor:
    """
    RandomForest với 200 trees.
    Input:  team_size, duration_weeks, risk_level (str), region (str)
    Output: estimated_cost + 95% confidence interval từ tree variance
    """

    def __init__(self):
        df = pd.read_csv(DATA_PATH)
        self.le = LabelEncoder()
        self.le.fit(REGIONS)

        # Build feature matrix (numpy array — no DataFrame warnings)
        X = np.column_stack([
            df["team_size"].values,
            df["duration_weeks"].values,
            df["risk_level"].values,
            self.le.transform(df["region"]),
        ])
        y = df["estimated_cost"].values

        self.rf = RandomForestRegressor(
            n_estimators=200,
            max_depth=8,
            min_samples_leaf=1,
            random_state=42,
        )
        self.rf.fit(X, y)

        # Feature importance (4 features)
        names = ["team_size", "duration_weeks", "risk_level", "region"]
        self.feature_importances = dict(zip(names, self.rf.feature_importances_.tolist()))

        # Model accuracy
        preds = self.rf.predict(X)
        self.train_r2 = r2_score(y, preds)
        print(f"[ML] RandomForest trained — R²={self.train_r2:.3f}, trees=200")

    def _encode_input(self, team_size, duration_weeks, risk_level, region):
        risk_num   = RISK_MAP.get(risk_level.lower(), 2)
        region_low = region.lower()
        if region_low not in REGIONS:
            region_low = "southeast"
        region_enc = self.le.transform([region_low])[0]
        return np.array([[team_size, duration_weeks, risk_num, region_enc]])

    def predict(self, team_size: int, duration_weeks: int,
                risk_level: str, region: str) -> dict:
        """
        Predict cost + 95% CI từ variance của 200 trees.
        Đây là kỹ thuật Conformal Prediction đơn giản.
        """
        X_pred = self._encode_input(team_size, duration_weeks, risk_level, region)

        # Collect predictions từ từng tree
        tree_preds = np.array([t.predict(X_pred)[0] for t in self.rf.estimators_])
        mean_cost  = float(np.mean(tree_preds))
        std_cost   = float(np.std(tree_preds))

        return {
            "estimated_cost":   round(mean_cost, 2),
            "confidence_low":   round(max(0, mean_cost - 1.96 * std_cost), 2),
            "confidence_high":  round(mean_cost + 1.96 * std_cost, 2),
            "std_dev":          round(std_cost, 2),
            "model_accuracy":   round(self.train_r2 * 100, 2),
        }

    def get_cost_breakdown(self, team_size, duration_weeks, risk_level, region) -> dict:
        """Phân bổ chi phí theo hạng mục (heuristic allocation)."""
        result = self.predict(team_size, duration_weeks, risk_level, region)
        total  = result["estimated_cost"]
        risk_n = RISK_MAP.get(risk_level.lower(), 2)
        risk_pct = 0.08 * risk_n / 3
        return {
            "total":          total,
            "labor":          round(total * 0.55, 2),
            "infrastructure": round(total * 0.20, 2),
            "risk_buffer":    round(total * risk_pct, 2),
            "overhead":       round(total * (1 - 0.55 - 0.20 - risk_pct), 2),
        }

    def sensitivity_analysis(self, base_team, base_weeks, base_risk, base_region) -> list[dict]:
        """
        Lab A Question: "BMI vs Smoker — which causes larger cost increase?"
        Tương đương: "team_size vs risk — cái nào quan trọng hơn?"
        """
        results = []
        for ts in [max(1, base_team - 4), base_team, base_team + 4, base_team + 8]:
            c = self.predict(ts, base_weeks, base_risk, base_region)["estimated_cost"]
            results.append({"variable": f"Team {ts}", "cost": c, "category": "team_size"})
        for wk in [max(1, base_weeks - 2), base_weeks, base_weeks + 2, base_weeks + 4]:
            c = self.predict(base_team, wk, base_risk, base_region)["estimated_cost"]
            results.append({"variable": f"{wk}w", "cost": c, "category": "duration"})
        for rl in ["low", "medium", "high"]:
            c = self.predict(base_team, base_weeks, rl, base_region)["estimated_cost"]
            results.append({"variable": rl.capitalize(), "cost": c, "category": "risk"})
        return results


# ── Quick test — Lab A scenario ───────────────────────
if __name__ == "__main__":
    pred = BudgetPredictor()

    # Simulate Lab A: Base profile
    base = pred.predict(12, 8, "medium", "southwest")
    print(f"\n--- Base Profile ---")
    print(f"Estimated Cost: ${base['estimated_cost']:,.2f}")
    print(f"95% CI: ${base['confidence_low']:,.0f} – ${base['confidence_high']:,.0f}")
    print(f"Model R²: {base['model_accuracy']}%")

    # Lab A Question: team_size +4 vs risk high

    print(f"Feature importances: {pred.feature_importances}")

"""
Executive Explainer Layer — Layer 5 của AI Agent Pipeline
Slide 7 — Role Prompting + Constrained Rewrite (bunq Finn pattern)

Chiến lược 2 lớp:
1. Deterministic report: luôn nhanh, luôn đúng sự thật
2. FLAN-T5 rewrite: load lazy, chỉ dùng khi user bật checkbox
"""

import threading
from datetime import datetime

URGENCY_ICON = {"high": "🔴", "medium": "🟡", "low": "🟢"}
DEPT_ICON    = {
    "Finance Advisor":    "💰",
    "IT Copilot":         "⚙️",
    "HR Bot":             "👥",
    "Product Management": "📦",
}

# ── FLAN-T5 lazy singleton ─────────────────────────────
_lock    = threading.Lock()
_tok     = None
_mdl     = None
_loaded  = False
_err_msg = None


def _load_flan() -> bool:
    """Load FLAN-T5-Large một lần duy nhất (lazy loading pattern)."""
    global _tok, _mdl, _loaded, _err_msg
    if _loaded:
        return True
    try:
        from transformers import AutoTokenizer, AutoModelForSeq2SeqLM
        print("[FLAN] Loading google/flan-t5-large … (first time ~60s)")
        _tok = AutoTokenizer.from_pretrained("google/flan-t5-large")
        _mdl = AutoModelForSeq2SeqLM.from_pretrained("google/flan-t5-large")
        _loaded = True
        print("[FLAN] ✅ Model ready")
        return True
    except Exception as e:
        _err_msg = str(e)
        print(f"[FLAN] ❌ {e}")
        return False


def _flan_rewrite(text: str) -> str | None:
    """
    Slide 7 — Constrained Rewrite + Role Prompting:
    "You are a compliance officer… rewrite in plain English…"
    """
    prompt = (
        "You are a senior compliance officer at a global bank. "
        "Rewrite the following operational alert as a professional executive brief. "
        "Rules: preserve all facts, no jargon, max 8 sentences, "
        "end with: 'Please consult a licensed professional before acting.'\n\n"
        f"Text:\n{text}"
    )
    with _lock:
        inputs = _tok(prompt, return_tensors="pt", truncation=True, max_length=512)
        out    = _mdl.generate(**inputs, max_new_tokens=280, temperature=0.3, do_sample=True)
        result = _tok.decode(out[0], skip_special_tokens=True).strip()
    return result if len(result) > 80 else None


class EnterpriseExplainer:

    def __init__(self):
        self.query_log = []

    def build_structured(self, business_name, issue_desc, kb_result, budget_result) -> str:
        """Layer 1: Deterministic report — luôn chính xác 100%."""
        dept     = kb_result.get("department", "Operations")
        urgency  = kb_result.get("urgency_level", "medium")
        solution = kb_result.get("solution", "N/A")
        action   = kb_result.get("recommended_action", "N/A")
        conf     = kb_result.get("confidence_score", 0.0)
        cost     = budget_result.get("estimated_cost", 0)
        accuracy = budget_result.get("model_accuracy", 0)
        ci_l     = budget_result.get("confidence_low", 0)
        ci_h     = budget_result.get("confidence_high", 0)
        ts       = datetime.now().strftime("%Y-%m-%d %H:%M")

        report = (
            f"=== EXECUTIVE OPERATIONAL ANALYSIS REPORT ===\n"
            f"Business: {business_name}\nGenerated: {ts}\n"
            f"{'─'*50}\n\n"
            f"{URGENCY_ICON.get(urgency,'🟡')} INCIDENT CLASSIFICATION\n"
            f"- Issue:   {issue_desc}\n"
            f"- Agent:   {DEPT_ICON.get(dept,'🏢')} {dept}\n"
            f"- Alert:   {urgency.upper()}\n"
            f"- Confidence: {conf*100:.1f}%\n\n"
            f"RECOMMENDED DIGITAL SOLUTION\n{solution}\n\n"
            f"IMMEDIATE ACTION\n{action}\n\n"
            f"BUDGET PROJECTION (ML Engine — RandomForest)\n"
            f"- Cost:    ${cost:,.2f} USD\n"
            f"- 95% CI:  ${ci_l:,.0f} – ${ci_h:,.0f}\n"
            f"- R²:      {accuracy:.1f}%\n\n"
            f"{'─'*50}\n"
            f"Note: Auto-generated per MIT PE Digital Transformation Framework."
        )
        self.query_log.append({
            "ts": ts, "business": business_name, "dept": dept,
            "urgency": urgency, "cost": cost, "conf": conf,
        })
        return report

    def build_executive_brief(
        self, business_name, issue_desc, kb_result, budget_result, use_llm=False
    ) -> tuple[str, str]:
        """
        Returns (report_text, flan_status_message).
        use_llm=False → deterministic only (fast)
        use_llm=True  → try FLAN-T5 rewrite, fallback gracefully
        """
        base = self.build_structured(business_name, issue_desc, kb_result, budget_result)

        if not use_llm:
            return base, "FLAN-T5 not activated — using deterministic report"

        ok = _load_flan()
        if not ok:
            return base, f"⚠️ FLAN load failed: {_err_msg}"

        try:
            rewritten = _flan_rewrite(base)
            if rewritten:
                combined = (
                    f"[✅ FLAN-T5-Large — Enhanced Executive Brief]\n{'─'*50}\n"
                    f"{rewritten}\n\n{'─'*50}\n"
                    f"[Original Deterministic Report]\n{base}"
                )
                return combined, "✅ FLAN-T5-Large rewrite applied"
            return base, "⚠️ FLAN output too short — fallback to deterministic"
        except Exception as e:
            return base, f"⚠️ FLAN error: {e}"

    def format_api_log(self, kb_result, budget_result) -> str:
        return (

"""
Orchestrator: kết nối tất cả các layers thành 1 pipeline
User Input → KB Search → ML Predict → Explainer → Result Dict
"""
import sys, os
sys.path.insert(0, os.path.dirname(__file__))

from core.database  import EnterpriseKnowledgeBase
from core.predictor import BudgetPredictor
from core.explainer import EnterpriseExplainer

# Singleton instances (load once)
_kb = _pred = _exp = None

def _init():
    global _kb, _pred, _exp
    if _kb is None:
        _kb   = EnterpriseKnowledgeBase()
        _pred = BudgetPredictor()
        _exp  = EnterpriseExplainer()
    return _kb, _pred, _exp


def run_analysis(
    business_name: str, admin_email: str,
    team_size: int, duration_weeks: int,
    risk_level: str, region: str,
    issue_description: str,
    use_llm: bool = False,
) -> dict:
    kb, pred, exp = _init()

    # Layer 1: Semantic Retrieval
    kb_results = kb.search(issue_description, top_k=1)
    kb_result  = kb_results[0] if kb_results else {
        "issue_pattern": issue_description,
        "solution": "Manual review required.",
        "department": "Operations",
        "urgency_level": "medium",
        "recommended_action": "Escalate to management.",
        "confidence_score": 0.0,
    }

    # Layer 2–3: ML Prediction + Breakdown
    budget_result  = pred.predict(team_size, duration_weeks, risk_level, region)
    cost_breakdown = pred.get_cost_breakdown(team_size, duration_weeks, risk_level, region)
    sensitivity    = pred.sensitivity_analysis(team_size, duration_weeks, risk_level, region)

    # Layer 4–5: Memory + Explainer
    executive_brief, flan_status = exp.build_executive_brief(
        business_name, issue_description, kb_result, budget_result, use_llm
    )
    api_log = exp.format_api_log(kb_result, budget_result)

    return {
        "executive_brief":    executive_brief,
        "flan_status":        flan_status,
        "api_log":            api_log,
        "kb_result":          kb_result,
        "budget_result":      budget_result,
        "cost_breakdown":     cost_breakdown,
        "sensitivity":        sensitivity,
        "feature_importances": pred.feature_importances,
        "kb_stats":           kb.get_stats(),
    }

def get_kb_stats(): kb, _, _ = _init(); return kb.get_stats()

def chart_sensitivity(sensitivity, p):
    cats = {"team_size":"Team Size", "duration":"Duration", "risk":"Risk Level"}
    fig = go.Figure()
    for i, (key, label) in enumerate(cats.items()):
        pts = [s for s in sensitivity if s["category"] == key]
        if not pts: continue
        fig.add_trace(go.Bar(name=label, x=[s["variable"] for s in pts],
                             y=[s["cost"] for s in pts], marker_color=p[i % 5]))
    fig.update_layout(title="Cost Sensitivity Analysis", barmode="group",
                      yaxis_tickprefix="$", yaxis_tickformat=",.0f",
                      paper_bgcolor="rgba(0,0,0,0)", plot_bgcolor="rgba(0,0,0,0)",
                      height=380, legend=dict(orientation="h", y=-0.22))
    return fig

def chart_pie(bd, p):
    fig = go.Figure(go.Pie(
        labels=["Labor","Infrastructure","Risk Buffer","Overhead"],
        values=[bd.get("labor",0), bd.get("infrastructure",0),
                bd.get("risk_buffer",0), bd.get("overhead",0)],
        marker_colors=p[:4], hole=0.42, textinfo="label+percent"))
    total = bd.get("total", 0)
    fig.update_layout(title="Cost Breakdown by Category", height=360,
                      annotations=[dict(text=f"${total:,.0f}", x=0.5, y=0.5,
                                        font_size=14, showarrow=False)],
                      paper_bgcolor="rgba(0,0,0,0)")
    return fig

def chart_feature_importance(fi, p):
    items  = sorted(fi.items(), key=lambda x: x[1])
    labels = [k.replace("_"," ").title() for k,_ in items]
    values = [v for _,v in items]
    fig = go.Figure(go.Bar(x=values, y=labels, orientation="h",
                           marker_color=p[:4], text=[f"{v:.3f}" for v in values],
                           textposition="outside"))
    fig.update_layout(title="RandomForest Feature Importance", height=300,
                      paper_bgcolor="rgba(0,0,0,0)", plot_bgcolor="rgba(0,0,0,0)")
    return fig

def chart_confidence(br, p):
    cost = br.get("estimated_cost", 0)
    ci_l = br.get("confidence_low", 0)
    ci_h = br.get("confidence_high", cost * 2)
    std  = max(br.get("std_dev", 1), 1)
    xs   = np.linspace(max(0, ci_l * 0.8), ci_h * 1.2, 200)
    ys   = np.exp(-0.5 * ((xs - cost) / std)**2)
    fig  = go.Figure()
    fig.add_trace(go.Scatter(x=xs, y=ys, fill="tozeroy",
                             fillcolor=p[0]+"30", line=dict(color=p[0], width=2)))
    fig.add_vline(x=cost, line=dict(color=p[1], width=2.5),
                  annotation_text=f"${cost:,.0f}", annotation_position="top right")
    for vx, lbl in [(ci_l, f"CI Low"), (ci_h, f"CI High")]:
        fig.add_vline(x=vx, line=dict(color=p[2], dash="dash"), annotation_text=lbl)
    fig.update_layout(title="Budget Prediction — 95% CI", height=340,
                      xaxis_tickprefix="$", xaxis_tickformat=",.0f",
                      paper_bgcolor="rgba(0,0,0,0)", plot_bgcolor="rgba(0,0,0,0)")
    return fig

def chart_scatter(p):

    df["risk_label"] = df["risk_level"].map({1:"Low",2:"Medium",3:"High"})

def render_chart(name, rj, theme):
    p = PALETTES.get(theme, list(PALETTES.values())[0])
    if not rj or rj == "{}":
        fig = go.Figure()
        fig.add_annotation(text="Click Analyze to generate charts",
                           xref="paper",yref="paper",x=0.5,y=0.5,
                           showarrow=False,font=dict(size=14,color="#888"))
        fig.update_layout(paper_bgcolor="rgba(0,0,0,0)",height=350)
        return fig
    r = json.loads(rj)
    bd = r.get("cost_breakdown",{}); sen = r.get("sensitivity",[])
    fi = r.get("feature_importances",{}); br = r.get("budget_result",{})
    ks = r.get("kb_stats",{}); urg = r.get("kb_result",{}).get("urgency_level","medium")
    if "Sensitivity" in name:   return chart_sensitivity(sen, p)
    if "Pie"         in name:   return chart_pie(bd, p)
    if "Feature"     in name:   return chart_feature_importance(fi, p)
    if "Confidence"  in name:   return chart_confidence(br, p)
    if "Donut"       in name:   return chart_donut(ks, urg, p)
    if "Scatter"     in name:   return chart_scatter(p)
    return go.Figure()


def analyze_all(bname, email, ts, dur, risk, region, issue, flan, chart, theme):
    if not bname.strip() or not issue.strip():
        msg = "⚠️ Please fill Business Name and Issue Description."
        return msg, "N/A", "", go.Figure(), "{}", "_No queries_", "—"

    r  = run_analysis(bname, email, int(ts), int(dur), risk, region, issue, flan)
    br = r["budget_result"]
    ml = (
        f"${br['estimated_cost']:,.2f} USD\n\n"
        f"95% CI: ${br['confidence_low']:,.0f} – ${br['confidence_high']:,.0f}\n"
        f"R²: {br['model_accuracy']:.1f}%   |   Std: ${br['std_dev']:,.0f}"
    )
    session_history.append({
        "Business": bname, "Dept": r["kb_result"]["department"],
        "Urgency": r["kb_result"]["urgency_level"].upper(),
        "Budget": f"${br['estimated_cost']:,.0f}",
        "Conf": f"{r['kb_result']['confidence_score']*100:.1f}%",
    })
    rj  = json.dumps(r, default=str)
    fig = render_chart(chart, rj, theme)
    hist_md = pd.DataFrame(session_history).to_markdown(index=False)
    return r["executive_brief"], ml, r["api_log"], fig, rj, hist_md, r["flan_status"]
